import React from 'react';
import {View, Text, TouchableOpacity, StyleSheet, Alert} from 'react-native';
import RazorpayCheckout from 'react-native-razorpay';

const HomeScreen = () => {
  const makePayment = () => {
    const options = {
      description: 'Credits towards consultation',
      image: require('../../assets/banner.jpg'), 
      currency: 'INR', 
      key: 'rzp_test_D1Oo29UXdKcJjW',  // Replace with your Razorpay Key ID
      amount: '50000', 
      name: 'Doctor Consultation', 
      prefill: {
        email: 'customer@example.com',
        contact: '9999999999',
        name: 'Anish Dhar', 
      },
      theme: {color: '#F37254'}, 
    };

    RazorpayCheckout.open(options)
      .then(data => {
        // On payment success
        Alert.alert('Success', `Payment ID: ${data.razorpay_payment_id}`);
      })
      .catch(error => {
        // On payment failure
        Alert.alert(
          'Payment Failed',
          `Error: ${error.code} | ${error.description}`,
        );
      });
  };

  return (
    <View style={styles.container}>
      <Text style={{fontWeight:'600', fontSize:30}}> â‚¹ 500</Text>
      <Text style={styles.header}>Make Payment</Text>
      <TouchableOpacity style={styles.payButton} onPress={makePayment}>
        <Text style={styles.payButtonText}>Pay Now</Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    backgroundColor: '#fff',
  },
  header: {
    fontSize: 24,
    fontWeight: 'bold',
    marginBottom: 20,
  },
  payButton: {
    backgroundColor: '#F37254',
    paddingVertical: 12,
    paddingHorizontal: 24,
    borderRadius: 6,
  },
  payButtonText: {
    color: '#fff',
    fontSize: 18,
  },
});
